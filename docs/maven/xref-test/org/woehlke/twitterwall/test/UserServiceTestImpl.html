55      @Override
56      public User createUser(String screenName) {
57          log.info("-----------------------------------------");
58          try {
59              log.info("screenName = "+ screenName);
60              User user = userService.findByScreenName(screenName);
61              log.info("userService.findByScreenName: found User = "+user.toString());
62              log.info("model.addAttribute user = "+user.toString());
63              return user;
64          } catch (EmptyResultDataAccessException e){
65              log.info("EmptyResultDataAccessException at userService.findByScreenName for screenName="+screenName);
66              TwitterProfile twitterProfile = twitterApiService.getUserProfileForScreenName(screenName);
67              log.info("twitterApiService.getUserProfileForScreenName: found TwitterProfile = "+twitterProfile.toString());
68              try {
69                  log.info("try: persistDataFromTwitter.storeUserProfile for twitterProfile = "+twitterProfile.toString());
70                  User user = persistDataFromTwitter.storeUserProfile(twitterProfile);
71                  log.info("persistDataFromTwitter.storeUserProfile: stored User = "+user.toString());
72                  log.info("model.addAttribute user = "+user.toString());
73                  return user;
74              } catch (EmptyResultDataAccessException ex){
75                  log.warn("persistDataFromTwitter.storeUserProfile raised EmptyResultDataAccessException: "+ex.getMessage());
76                  User user = getDummyUser();
77                  log.info("model.addAttribute user = "+user.toString());
78                  return user;
79              } catch (NoResultException exe) {
80                  log.warn("persistDataFromTwitter.storeUserProfile raised NoResultException: "+exe.getMessage());
81                  User user = getDummyUser();
82                  log.info("model.addAttribute user = "+user.toString());
83                  return user;
84              }
85          }  finally {
86              log.info("... finally done ...");
87              log.info("-----------------------------------------");
88          }
89      }
