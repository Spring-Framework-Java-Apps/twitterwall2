4   import org.woehlke.twitterwall.oodm.entities.common.DomainObjectWithUrl;
5   
6   import javax.persistence.*;
7   
8   /**
9    * Created by tw on 23.06.17.
10   */
11  @Entity
12  @Table(name = "url_cache",
13          uniqueConstraints = { @UniqueConstraint(name="unique_url_cache", columnNames = {"url"})
14  }, indexes = {
15          @Index(name="idx_url_cache_expanded", columnList="expanded")
16  })
17  @NamedQueries({
18          @NamedQuery(
19                  name = "UrlCache.findByUrl",
20                  query = "select t from UrlCache as t where t.url=:url"
21          )
22  })
23  public class UrlCache implements DomainObjectWithUrl<UrlCache> {
24      
25      private static final long serialVersionUID = 1L;
26  
27      @Id
28      @GeneratedValue(strategy = GenerationType.AUTO)
29      private Long id;
30  
31      @Column(length=2048)
32      private String expanded;
33  
34      @Column(nullable = false,length=1024)
35      private String url;
36  
37      @Transient
38      public boolean isUrlAndExpandedTheSame(){
39         return  url.compareTo(expanded) == 0;
40      }
41  
42      public Long getId() {
43          return id;
44      }
45  
46      public void setId(Long id) {
47          this.id = id;
48      }
49  
50      public String getExpanded() {
51          return expanded;
52      }
53  
54      public void setExpanded(String expanded) {
55          this.expanded = expanded;
56      }
57  
58      public String getUrl() {
59          return url;
60      }
61  
62      public void setUrl(String url) {
63          this.url = url;
64      }
65  
66      @Override
67      public boolean equals(UrlCache o) {
68          if (this == o) return true;
69          if (!(o instanceof UrlCache)) return false;
70  
71          UrlCache urlCache = (UrlCache) o;
72  
73          return url.equals(urlCache.url);
74      }
75  
76      @Override
77      public int hashCode() {
78          return url.hashCode();
79      }
80  
81      @Override
82      public int compareTo(UrlCache other) {
83          return url.compareTo(other.getUrl());
84      }
85  
86      @Override
87      public String toString() {
88          return "UrlCache{" +
89                  "id=" + id +
90                  ", expanded='" + expanded + '\'' +
91                  ", url='" + url + '\'' +
92                  '}';
93      }
94  }
