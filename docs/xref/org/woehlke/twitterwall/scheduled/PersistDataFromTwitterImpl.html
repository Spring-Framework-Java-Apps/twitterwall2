63      @Override
64      public Tweet storeOneTweet(org.springframework.social.twitter.api.Tweet myTweet) {
65          Tweet tweet = tweetTransformService.transform(myTweet);
66          tweet = this.storeOneTweetPerform(tweet);
67          return tweet;
68      }
69  
70      /** private Method because of recursive Method Call in this Method **/
71      private Tweet storeOneTweetPerform(Tweet tweet){
72          /** Retweeted Tweet */
73          Tweet retweetedStatus = tweet.getRetweetedStatus();
74          if (retweetedStatus != null) {
75              /** private Method because of recursive Method Call in this Method **/
76              retweetedStatus = this.storeOneTweetPerform(retweetedStatus);
77              tweet.setRetweetedStatus(retweetedStatus);
78          }
79          /** The User */
80          User user = tweet.getUser();
81          user = userService.storeUserProcess(user);
82          tweet.setUser(user);
83          /** The Entities */
84          Set<Url> urls = new LinkedHashSet<>();
85          Set<HashTag> tags = new LinkedHashSet<HashTag>();
86          Set<Mention> mentions = new LinkedHashSet<Mention>();
87          Set<Media> media = new LinkedHashSet<Media>();
88          Set<TickerSymbol> tickerSymbols = new LinkedHashSet<TickerSymbol>();
89          for (TickerSymbol tickerSymbol : tweet.getTickerSymbols()) {
90              tickerSymbols.add(tickerSymbolService.storeTickerSymbol(tickerSymbol));
91          }
92          for (Mention mention : tweet.getMentions()) {
93              mentions.add(mentionService.store(mention));
94          }
95          for (Media medium : tweet.getMedia()) {
96              media.add(mediaService.store(medium));
97          }
98          for (HashTag hashTag : tweet.getTags()) {
99              tags.add(hashTagService.store(hashTag));
100         }
101         for (Url url : tweet.getUrls()) {
102             urls.add(urlService.getPersistentUrlFor(url.getUrl()));
103         }
104         tweet.setUrls(urls);
105         tweet.setTags(tags);
106         tweet.setMentions(mentions);
107         tweet.setMedia(media);
108         tweet.setTickerSymbols(tickerSymbols);
109         /** Tweet itself */
110         tweet = tweetService.store(tweet);
111         return tweet;
112     }
