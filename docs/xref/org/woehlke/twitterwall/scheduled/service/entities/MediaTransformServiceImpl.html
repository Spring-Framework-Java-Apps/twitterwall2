10   * Created by tw on 28.06.17.

12  @Service
13  @Transactional(propagation = Propagation.REQUIRED, readOnly = true)
14  public class MediaTransformServiceImpl implements MediaTransformService {
15      
16      @Override
17      public Media transform(MediaEntity medium) {
18          long idTwitter = medium.getId();
19          String mediaHttp = medium.getMediaUrl();
20          String mediaHttps = medium.getMediaSecureUrl();
21          String url = medium.getUrl();
22          String display = medium.getDisplayUrl();
23          String expanded = medium.getExpandedUrl();
24          String type = medium.getType();
25          int[] indices = medium.getIndices();
26          Media myMediaEntity = new Media(idTwitter, mediaHttp, mediaHttps, url, display, expanded, type, indices);
27          return myMediaEntity;
28      }
29  }
