20  @ControllerAdvice
21  public class GlobalExceptionHandler extends AbstractTwitterwallController {
22  
23      private static final Logger log = LoggerFactory.getLogger(GlobalExceptionHandler.class);
24  
25      @Value("${twitterwall.frontend.menu.appname}")
26      private String menuAppName;
27  
28      @Value("${twitter.searchQuery}")
29      private String searchterm;
30  
31      @Value("${twitterwall.frontend.info.webpage}")
32      private String infoWebpage;
33  
34      @Value("${twitterwall.frontend.theme}")
35      private String theme;
36  
37      @Value("${twitterwall.context.test}")
38      private boolean contextTest;
39  
40      @Value("${twitterwall.frontend.imprint.screenName}")
41      private String imprintScreenName;
42  
43      @Value("${twitterwall.frontend.idGoogleAnalytics}")
44      private String idGoogleAnalytics;
45  
46  
47      private final TwitterApiService twitterApiService;
48  
49      private final PersistDataFromTwitter persistDataFromTwitter;
50  
51      @Autowired
52      public GlobalExceptionHandler(TwitterApiService twitterApiService, PersistDataFromTwitter persistDataFromTwitter) {
53          this.twitterApiService = twitterApiService;
54          this.persistDataFromTwitter = persistDataFromTwitter;
55      }
56  
57      @ExceptionHandler(IllegalArgumentException.class)
58      public ModelAndView handleIllegalArgumentException(HttpServletRequest request, Exception ex) {
59          log.warn("IllegalArgumentException occured :: URL=" + request.getRequestURL());
60          log.warn(ex.getMessage());
61          return getTemplate(request, ex);
62      }
63      
64      private ModelAndView getTemplate(HttpServletRequest request, Exception ex) {
65          ModelAndView mav = new ModelAndView();
66          String symbol = Symbols.EXCEPTION.toString();
67          String title = "Exception";
68          String subtitle = ex.getMessage();
69          mav = super.setupPage(mav, title, subtitle, symbol);
70          mav.addObject("exception", ex);
71          mav.addObject("url", request.getRequestURL());
72          mav.setViewName("persistentObjectNotFound");
73          return mav;
74      }
75  
76      @Override
77      public void afterPropertiesSet() throws Exception {
78          super.setupAfterPropertiesSetWithTesting(twitterApiService,persistDataFromTwitter,menuAppName,searchterm,infoWebpage,theme,contextTest,imprintScreenName,idGoogleAnalytics);
79      }
80  }
